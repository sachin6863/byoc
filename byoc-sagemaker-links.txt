create the docker file with base image (Dockerfile name should be start with Capital D)

aws ecr create-repository \
    --repository-name hf-pytorch-cpu \
    --image-scanning-configuration scanOnPush=false \
    --region us-east-1

aws ecr get-login-password --region $AWS_DEFAULT_REGION | docker login --username AWS --password-stdin 763104351884.dkr.ecr.$AWS_DEFAULT_REGION.amazonaws.com
docker build -t hf-pytorch-cpu:1.0 .
docker tag hf-pytorch-cpu:1.0 839052460858.dkr.ecr.us-east-1.amazonaws.com/hf-pytorch-cpu:1.0
aws ecr get-login-password --region $AWS_DEFAULT_REGION | docker login --username AWS --password-stdin 839052460858.dkr.ecr.$AWS_DEFAULT_REGION.amazonaws.com
docker push 839052460858.dkr.ecr.us-east-1.amazonaws.com/hf-pytorch-cpu:1.0
